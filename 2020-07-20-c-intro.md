__The assignment:__

_Write a program that computes the total sales tax on a $95 purchase.
Assume the state sales tax is 6.5 percent and the county sales tax is 2 percent.
Display the purchase price, state tax, county tax, and total tax amounts on the screen._

__My Solution:__

{% highlight c %}

#include <iostream>
using namespace std;

int main()
{
	// declare and initialize purchase price and tax rates
	double stateTaxRate = 0.065, countyTaxRate = 0.02, 
	double purchasePrice = 95.0;

	// declare and initialize variables to hold calculated tax
	double stateTaxOnPurchase = purchasePrice * stateTaxRate;
	double countyTaxOnPurchase = purchasePrice * countyTaxRate;
	double totalTax = stateTaxOnPurchase + countyTaxOnPurchase;

	// output to console
	cout << "Purchase Price: " << purchasePrice << endl;
	cout << "State Tax: " << stateTaxOnPurchase << endl;
	cout << "County Tax : " << countyTaxOnPurchase << endl;
	cout << "Total Tax: " << totalTax << endl;

	return 0;
}
{% endhighlight %}
